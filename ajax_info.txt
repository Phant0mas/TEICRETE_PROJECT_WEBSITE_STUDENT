<script language="javascript">
    function changeImage() {

        if (document.getElementById("image").src == "content/images/beethoven_1.jpg") 
        {
            document.getElementById("image").src = "content/images/beethoven_2.gif";
        }
        else 
        {
            document.getElementById("image").src = "content/images/beethoven_1.jpg";
        }
    }
</script>

<img src="content/images/beethoven_2.gif" style="height: 640px; width: 480px" onclick="changeImage()" id="image" />
